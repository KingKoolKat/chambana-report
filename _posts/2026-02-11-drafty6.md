<script>
  (function () {
    const root = document.getElementById("drafty-connections-game");
    if (!root || root.dataset.bound === "1") return;
    root.dataset.bound = "1";

    const groups = [
      { title: "NATO ALPHABET", words: ["ECHO", "ROMEO", "DELTA", "CHARLIE"], color: "var(--dc-purple)" },
      { title: "SHAWN MENDES SONGS", words: ["MERCY", "STITCHES", "SENORITA", "LOST IN JAPAN"], color: "var(--dc-blue)" },
      { title: "FEUDS OF ALL SORTS", words: ["VENDETTA", "RIVALRY", "CLASH", "QUARREL"], color: "var(--dc-green)" },
      { title: "PALINDROMES", words: ["ANA", "MADAM", "RADAR", "KAYAK"], color: "var(--dc-yellow)" }
    ];
    const colorToEmoji = {
      "var(--dc-yellow)": "🟨",
      "var(--dc-green)": "🟩",
      "var(--dc-blue)": "🟦",
      "var(--dc-purple)": "🟪"
    };
    const wordToGroup = new Map();
    groups.forEach((group) => {
      group.words.forEach((word) => {
        wordToGroup.set(word, group);
      });
    });

    const grid = root.querySelector(".dc-grid");
    const solvedBox = root.querySelector(".dc-solved");
    const message = root.querySelector(".dc-message");
    const dots = Array.from(root.querySelectorAll(".dc-dot"));
    const buttons = {
      submit: root.querySelector("[data-action='submit']"),
      clear: root.querySelector("[data-action='clear']"),
      shuffle: root.querySelector("[data-action='shuffle']"),
      share: root.querySelector("[data-action='share']")
    };

    let selected = new Set();
    let solved = new Set();
    let guesses = [];
    let mistakes = 0;
    let over = false;
    let words = [];

    function shuffle(arr) {
      for (let i = arr.length - 1; i > 0; i -= 1) {
        const j = Math.floor(Math.random() * (i + 1));
        [arr[i], arr[j]] = [arr[j], arr[i]];
      }
      return arr;
    }

    function initialize() {
      selected = new Set();
      solved = new Set();
      guesses = [];
      mistakes = 0;
      over = false;
      words = shuffle(groups.flatMap((g) => g.words.slice()));
      renderSolved();
      updateMistakes();
      setMessage("Select 4 words, then submit.");
      renderGrid();
      updateButtons();
    }

    function setMessage(text) {
      message.textContent = text;
    }

    function updateMistakes() {
      dots.forEach((dot, i) => {
        dot.classList.toggle("dc-used", i < mistakes);
      });
    }

    function renderSolved() {
      const solvedGroups = groups.filter((g) => g.words.every((w) => solved.has(w)));
      solvedBox.innerHTML = solvedGroups
        .map((g) => (
          '<div class="dc-solved-group" style="background:' + g.color + '">' +
            '<div class="dc-solved-title">' + g.title + '</div>' +
            '<div class="dc-solved-words">' + g.words.join(", ") + "</div>" +
          "</div>"
        ))
        .join("");
    }

    function renderGrid() {
      const remaining = words.filter((w) => !solved.has(w));
      grid.innerHTML = "";
      remaining.forEach((word) => {
        const btn = document.createElement("button");
        btn.type = "button";
        btn.className = "dc-tile";
        if (selected.has(word)) btn.classList.add("dc-selected");
        btn.textContent = word;
        btn.setAttribute("role", "gridcell");
        btn.disabled = over;
        btn.addEventListener("click", () => toggleWord(word));
        grid.appendChild(btn);
      });
    }

    function toggleWord(word) {
      if (over) return;
      if (selected.has(word)) {
        selected.delete(word);
      } else {
        if (selected.size >= 4) return;
        selected.add(word);
      }
      renderGrid();
      updateButtons();
    }

    function updateButtons() {
      buttons.submit.disabled = over || selected.size !== 4;
      buttons.clear.disabled = over || selected.size === 0;
      buttons.shuffle.disabled = over;
      buttons.share.disabled = !over || guesses.length === 0;
    }

    function guessToEmojiRow(guessWords) {
      return guessWords
        .map((word) => {
          const group = wordToGroup.get(word);
          return group ? colorToEmoji[group.color] : "⬜";
        })
        .join("");
    }

    function buildShareText() {
      const rows = guesses.map((guess) => guessToEmojiRow(guess));
      const header = "Drafty Connections Completed (" + mistakes + " mistakes):";
      return [header, "", ...rows].join("\n");
    }

    async function copyToClipboard(text) {
      if (navigator.clipboard && window.isSecureContext) {
        await navigator.clipboard.writeText(text);
        return;
      }

      const hiddenInput = document.createElement("textarea");
      hiddenInput.value = text;
      hiddenInput.setAttribute("readonly", "");
      hiddenInput.style.position = "absolute";
      hiddenInput.style.left = "-9999px";
      document.body.appendChild(hiddenInput);
      hiddenInput.select();
      const copied = document.execCommand("copy");
      document.body.removeChild(hiddenInput);
      if (!copied) {
        throw new Error("Clipboard copy failed");
      }
    }

    async function shareResults() {
      if (!over || guesses.length === 0) return;
      try {
        await copyToClipboard(buildShareText());
        setMessage("Results copied to clipboard.");
      } catch (error) {
        setMessage("Could not copy results.");
      }
    }

    function clearSelection() {
      if (over) return;
      selected.clear();
      renderGrid();
      updateButtons();
      setMessage("Selection cleared.");
    }

    function shuffleBoard() {
      if (over) return;
      const unsolved = words.filter((w) => !solved.has(w));
      const reshuffled = shuffle(unsolved.slice());
      let idx = 0;
      words = words.map((w) => (solved.has(w) ? w : reshuffled[idx++]));
      renderGrid();
      setMessage("Board shuffled.");
    }

    function findGroup(wordSet) {
      return groups.find((g) => g.words.every((w) => wordSet.has(w)));
    }

    function isOneAway(wordSet) {
      return groups.some((g) => {
        const hits = g.words.filter((w) => wordSet.has(w)).length;
        return hits === 3;
      });
    }

    function submitSelection() {
      if (selected.size !== 4 || over) return;

      const guessWords = Array.from(selected);
      guesses.push(guessWords);
      const choice = new Set(selected);
      const matchedGroup = findGroup(choice);

      if (matchedGroup) {
        matchedGroup.words.forEach((w) => solved.add(w));
        selected.clear();
        renderSolved();
        renderGrid();
        updateButtons();

        if (solved.size === 16) {
          over = true;
          updateButtons();
          setMessage("Perfect. You found all 4 groups. Use Share to copy results.");
          return;
        }

        setMessage("Correct: " + matchedGroup.title + ".");
        return;
      }

      mistakes += 1;
      updateMistakes();
      setMessage(isOneAway(choice) ? "One away..." : "Not a group.");
      selected.clear();
      renderGrid();
      updateButtons();

      if (mistakes >= 4) {
        over = true;
        groups.forEach((g) => g.words.forEach((w) => solved.add(w)));
        renderSolved();
        renderGrid();
        updateButtons();
        setMessage("Game over. Use Share to copy results.");
      }
    }

    buttons.submit.addEventListener("click", submitSelection);
    buttons.clear.addEventListener("click", clearSelection);
    buttons.shuffle.addEventListener("click", shuffleBoard);
    buttons.share.addEventListener("click", shareResults);

    initialize();
  })();
</script>
