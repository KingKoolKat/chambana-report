    function buildConicBackground(items, palette) {
      const slice = 360 / items.length;
      const stops = items
        .map((_, idx) => {

      });

    function readSavedResult() {
      try {
        const raw = window.localStorage.getItem(storageKey);
        if (!raw) return null;
        const parsed = JSON.parse(raw);
        if (!parsed || typeof parsed.primaryIndex !== "number") return null;
        const prize = primaryPrizes[parsed.primaryIndex];
        if (!prize || !prize.qualifierPool) return null;
        const expectedPool = prize.qualifierPool;
        const expectedNames = qualifierPools[expectedPool];
        if (!expectedNames || expectedNames.length === 0) return null;
        if (parsed.qualifierPool !== expectedPool) return null;
        if (typeof parsed.qualifierIndex !== "number") return null;
        if (parsed.qualifierIndex < 0 || parsed.qualifierIndex >= expectedNames.length) return null;

        const resolvedQualifierText = expectedNames[parsed.qualifierIndex];
        const resolvedFinalText = formatPrizeText(prize.label, resolvedQualifierText);

        if (parsed.qualifierText && parsed.qualifierText !== resolvedQualifierText) return null;
        if (parsed.finalText && parsed.finalText !== resolvedFinalText) return null;

        parsed.qualifierText = resolvedQualifierText;
        parsed.finalText = resolvedFinalText;
        return parsed;
      } catch (error) {
        return null;
      }
    }

    function calculateNextRotation(current, itemsLength, winnerIndex, minTurns) {
      const segment = 360 / itemsLength;
      const targetAngle = (360 - (winnerIndex * segment + segment / 2)) % 360;
      const currentNorm = ((current % 360) + 360) % 360;
      const deltaToTarget = (targetAngle - currentNorm + 360) % 360;
      return current + minTurns * 360 + deltaToTarget;
    }

      hasSpun = true;
      isSpinning = false;

      const primaryPrize = primaryPrizes[primaryIndex];
      lockSpinButton(primaryPrize.label);
      showPrimaryWheel();
      showPrizeModal(formatPrizeText(primaryPrize.label, ""), "", true);

      saveResult({
        spunAt: new Date().toISOString(),
        primaryIndex: primaryIndex,
        primaryText: primaryPrize.label
      });
    }

    function spinQualifier(primaryIndex, qualifierPoolKey) {
      const names = qualifierPools[qualifierPoolKey];
      if (!names || names.length === 0) {
        finalizeWithoutQualifier(primaryIndex);
        return;
      }

      currentQualifierPool = qualifierPoolKey;
      showQualifierWheel();
      renderLabels(secondaryLabelsEl, secondaryShellEl, names, "dws-label-secondary", 0.35);
      secondaryWheelEl.style.background = buildConicBackground(names, colorSet);

      const forcedWinner = forcedQualifierWinner[qualifierPoolKey];
      const forcedIndex = names.indexOf(forcedWinner);
      const qualifierIndex = forcedIndex >= 0 ? forcedIndex : randomIndex(names.length);
      secondaryRotation = calculateNextRotation(secondaryRotation, names.length, qualifierIndex, 8);

      const onQualifierDone = function () {
        secondaryWheelEl.removeEventListener("transitionend", onQualifierDone);

        const qualifierText = names[qualifierIndex];
        const primaryPrize = primaryPrizes[primaryIndex];
        const finalText = formatPrizeText(primaryPrize.label, qualifierText);

        showPrizeModal(finalText, qualifierText, true);
        lockSpinButton(finalText);
        hasSpun = true;
        isSpinning = false;

        saveResult({
          spunAt: new Date().toISOString(),
          primaryIndex: primaryIndex,
          primaryText: primaryPrize.label,
          qualifierPool: qualifierPoolKey,
          qualifierIndex: qualifierIndex,
          qualifierText: qualifierText,
          finalText: finalText
        });
      };

      secondaryWheelEl.addEventListener("transitionend", onQualifierDone);
      requestAnimationFrame(function () {
        applyRotation(secondaryWheelEl, secondaryLabelsEl, secondaryRotation);
      });
    }

    function spinPrimary() {
      if (hasSpun || isSpinning) return;

      isSpinning = true;
      spinButton.disabled = true;
      closePrizeModal();
      showPrimaryWheel();

      const primaryIndex = isJonHugOverride
        ? primaryIndexByQualifierPool.hug
        : riggedPrimaryIndices.length
          ? riggedPrimaryIndices[randomIndex(riggedPrimaryIndices.length)]
          : randomIndex(primaryPrizes.length);
      primaryRotation = calculateNextRotation(primaryRotation, primaryPrizes.length, primaryIndex, 12);

      const onPrimaryDone = function () {
        primaryWheelEl.removeEventListener("transitionend", onPrimaryDone);

        const prize = primaryPrizes[primaryIndex];
        if (prize.qualifierPool) {
          showPrizeModal("You won \"" + prize.label + "\". Let's see who it is with...", "", false);
          setTimeout(function () {
            closePrizeModal();
            spinQualifier(primaryIndex, prize.qualifierPool);
          }, 3000);
          return;
        }

        isSpinning = false;
        finalizeWithoutQualifier(primaryIndex);
      };

      primaryWheelEl.addEventListener("transitionend", onPrimaryDone);
      requestAnimationFrame(function () {
        applyRotation(primaryWheelEl, primaryLabelsEl, primaryRotation);
      });
    }

    function initializeWheels() {
      primaryWheelEl.style.background = buildConicBackground(primaryPrizes, colorSet);
      renderLabels(
        primaryLabelsEl,
        primaryShellEl,
        primaryPrizes.map(function (item) {
          return item.label;
        }),
        "dws-label-primary",
        0.365
      );

      const currentSecondary = qualifierPools[currentQualifierPool] || qualifierPools.pickleball;
      secondaryWheelEl.style.background = buildConicBackground(currentSecondary, colorSet);
      renderLabels(secondaryLabelsEl, secondaryShellEl, currentSecondary, "dws-label-secondary", 0.35);
    }

    initializeWheels();

    window.addEventListener("resize", function () {
      initializeWheels();
      applyRotation(primaryWheelEl, primaryLabelsEl, primaryRotation);
      applyRotation(secondaryWheelEl, secondaryLabelsEl, secondaryRotation);
    });

    if (shouldResetSavedSpin) {
      clearSavedResult();
    }

    const saved = readSavedResult();
    if (saved) {
      showSavedState(saved);
    } else {
      showPrimaryWheel();
      unlockSpinButtonForFirstUse();
    }

    spinButton.addEventListener("click", spinPrimary);
    closeModalButton.addEventListener("click", closePrizeModal);
    modalEl.addEventListener("click", function (event) {
      if (event.target === modalEl) {
        closePrizeModal();
      }
    });
    document.addEventListener("keydown", function (event) {
      if (event.key === "Escape" && !modalEl.hidden) {
        closePrizeModal();
      }
    });
  })();
</script>
